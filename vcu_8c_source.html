40int VCUHandlerStep()
41{
42 if (calibrationModeStatus == OK_STATUS){
43 //BMS FAULT
44 bmsFault = GetParameterValue(rxParameters[0]);
45 if (bmsFault > 0){ // if it is 0 BMS encountered some fault
46 SetDigitalOutputState(digitalInterfaces[0], GPIO_PIN_RESET); //we will have to change this PIN but it probably handles some relay logic by setting pin to LOW
47 }
48 else{
49 SetDigitalOutputState(digitalInterfaces[0], GPIO_PIN_SET);//same here but HIGH
50 }
51
52 timerCounter = __HAL_TIM_GetCounter(timer);
53
54 if ((timerCounter - lastValue) < 0)
55 {
56 timerOverflow += 1;
57 }
58
59 lastValue = timerCounter;
60
61 totalMillis = timerCounter + timerOverflow * 65535;
62
63 ReadAnalogSensorList(sensors);
64
65 //APPS1 FILTERING
66 currentSensorRead = (1 - ANALOG_FILTER_CONST) * previousSensorRead + ANALOG_FILTER_CONST * sensors[APPS1_SENSOR].readValue;
67 previousSensorRead = currentSensorRead;
68 //APPS1 FILTERING END
69
70 UpdatePowertrainControlParameters(); //APPS1 % calculation
71 GetTorqueDemand(currApps1Percentage); //torque demand calculation
72 delta = 1;
73 delta = floor(sensors[APPS1_SENSOR].value) - floor(sensors[APPS2_SENSOR].value);
74
75 if (sensors[APPS1_SENSOR].value < 0 || sensors[APPS2_SENSOR].value < 0){
76 txParameters[TORQUE_DEMAND].parameterValue = 0;
77 txParameters[DRIVE_SWITCH].parameterValue = 0;
78 PackAndTransmitDataOverCan();
79 return ERROR_STATUS;
80 }
81
82 if(delta > 200 || delta < -200 ){
83
84 if (isPedalError != PEDAL_ERROR){
85 lastPedalErrorTime = totalMillis;
86 isPedalError = PEDAL_ERROR;
87 }
88
89 if (totalMillis - lastPedalErrorTime > 100){
90 txParameters[TORQUE_DEMAND].parameterValue = 0;
91 txParameters[DRIVE_SWITCH].parameterValue = 0;
92 }
93
94 PackAndTransmitDataOverCan();
95 return OK_STATUS;
96 }
97 else{
98 isPedalError = OKAY;
99 }
100
101 if (sensors[APPS1_SENSOR].value < 20){
102 txParameters[TORQUE_DEMAND].parameterValue = 0;
103 }
104
105 if (sensors[APPS1_SENSOR].value > 20){
106 txParameters[DRIVE_SWITCH].parameterValue = 1;
107 }
108 else{
109 txParameters[DRIVE_SWITCH].parameterValue = 0;
110 }
111
112 PackAndTransmitDataOverCan();
113 }

30int TxCanMsg(FDCAN_HandleTypeDef canChannel, _canMessage* msg) {
31 // Configure the CAN message header

34
35 // Attempt to add the message to the transmit FIFO queue
36 if (HAL_FDCAN_AddMessageToTxFifoQ(&canChannel, &TxHeader, msg->data) != HAL_OK) {

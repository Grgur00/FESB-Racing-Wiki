21float GetParameterValue(_parameter parameter) {
22 float value = 0; // Initialize the value to 0
23
24 // If the parameter length is 1, return the raw byte value directly
25 if (parameter.dParams.length == 1) {
26 value = parameter.rawBytes[0];
27 return value;
28 }
29
30 // Calculate the value based on endianness
31 if (parameter.dParams.endian == INTEL) {
32 // Intel (little-endian) format: least significant byte first
33 for (int i = 0; i < parameter.dParams.length; i++) {
34 value += parameter.rawBytes[i] << (8 * i);
35 }
36 } else {
37 // Motorola (big-endian) format: most significant byte first
38 int start = 8 * parameter.dParams.length;
39 value += parameter.rawBytes[0] << (8);
40 for (int i = 1; i < parameter.dParams.length; i++) {
41 value += parameter.rawBytes[i] << (start - 8 * (i + 1));
42 }
43 }
44
45 // Apply the scaling factor and return the final value
46 return (value * parameter.dParams.factor);
47}

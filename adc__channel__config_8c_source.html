98float GetSensorValue(_sensorInterface* sensor) {
99 // Check if the read value exceeds the maximum limit
100 if (sensor->readValue > sensor->maxRead) {
101 sensor->faultStatus = SENSOR_STATUS_FAULT; // Set fault status
102 return -1; // Return error value
103 }
104
105 // Check if the read value is below the minimum limit
106 if (sensor->readValue < sensor->minRead) {
107 sensor->faultStatus = SENSOR_STATUS_FAULT; // Set fault status
108 return -1; // Return error value
109 }
110
111 // Calculate the percentage of the sensor value within the defined range
112 float percentage = ((float)(sensor->readValue - sensor->minRead) / (sensor->maxRead - sensor->minRead));
113 sensor->faultStatus = SENSOR_STATUS_OK; // Set fault status to OK
114 return percentage; // Return the calculated percentage
115}

126uint8_t* PackData(_sensorInterface* interface) {
127 // Check if the size of the data is 2 bytes
128 if (interface->params.size == 2) {
129 interface->value *= interface->params.factor; // Scale the value
130 interface->value += interface->params.offset; // Apply the offset
131 uint32_t signalVal32 = floor(interface->value); // Floor the value to an integer
132 uint16_t castValue = (uint16_t)signalVal32; // Cast to 16-bit value
133
134 // Pack the high and low bytes of the value into hexData
135 interface->hexData[0] = (castValue >> 8) & 0xFF; // High byte
136 interface->hexData[1] = castValue & 0xFF; // Low byte
137
138 // Store the packed data into the message structure
139 interface->msg->data[interface->params.index] = interface->hexData[0];
140 interface->msg->data[interface->params.index + 1] = interface->hexData[1];
141 } else {
142 // If size is not 2, cast the value to 8-bit and store it
143 uint8_t castValue = (uint8_t)interface->value;
144 interface->msg->data[interface->params.index] = castValue; // Store the value in the message
145 }
146
147 return interface->hexData; // Return the packed hex data
148}

160int SelectAdcChannel(_sensorInterface* interface, uint32_t channel, uint8_t activate) {

163 sConfig.SamplingTime = ADC_SAMPLETIME_1CYCLE_5; // Set the sampling time

165 sConfig.OffsetNumber = ADC_OFFSET_NONE; // No offset

168
169 // Set the rank based on whether the channel is being activated or deactivated

172 } else {
173 sConfig.Rank = ADC_REGULAR_RANK_4; // Set rank for inactive channel
174 }
175
176 HAL_ADC_ConfigChannel(&interface->adcHandle, &sConfig); // Configure the ADC channel
177
178 return OK_STATUS; // Return success status
